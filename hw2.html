from sklearn.ensemble import RandomForestRegressor
# deal with missing values in age
def set_missing_age(df):
    age_df = df[["Age","Pclass","SibSp","Parch","Fare"]]
    # divide the parts with missing value and without missing value
    age_known = age_df[age_df.Age.notnull()]
    age_unknown = age_df[age_df.Age.isnull()]
    # divide feature and targets in the known part
    y = age_known.iloc[:,0]
    X = age_known.iloc[:,1:]
    # fit to the random forest
    regr = RandomForestRegressor(random_state=0,n_estimators=100)
    regr.fit(X,y)
    predict_age = regr.predict(age_unknown.iloc[:,1:])
    df.loc[df.Age.isnull(),"Age"] = predict_age
    return df
set_missing_age(train)
set_missing_age(test)

# extract title from name
import re
for data in all_data:
    name = data['Name']
    title = []
    for nm in name:
        ti = re.split(r'[,.]',nm)[1]
        ti = ti.replace(' ','')
        title.append(ti)
    data['Title'] = title
